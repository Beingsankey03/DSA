class Solution {
    public int singleNumber(int[] nums) {
        int ones = 0, twos = 0;

        for (int num : nums) {
            // First update 'ones'
            ones = (ones ^ num) & ~twos;

            // Then update 'twos'
            twos = (twos ^ num) & ~ones;
        }

        // At the end, 'ones' contains the unique element
        return ones;
    }
}
