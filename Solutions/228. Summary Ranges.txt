import java.util.*;

class Solution {
    public List<String> summaryRanges(int[] nums) {
        List<String> result = new ArrayList<>();
        int n = nums.length;
        
        if (n == 0) return result;  // edge case: empty array

        int start = 0; // pointer to track the beginning of a range

        for (int i = 0; i < n; i++) {
            // check if this is the last element OR nums[i] and nums[i+1] are not consecutive
            if (i == n - 1 || nums[i] + 1 != nums[i + 1]) {
                if (start == i) {
                    // single element range
                    result.add(String.valueOf(nums[start]));
                } else {
                    // range with more than one element
                    result.add(nums[start] + "->" + nums[i]);
                }
                // update start for next range
                start = i + 1;
            }
        }
        return result;
    }
}
