class Solution {
    public int maxProfit(int[] prices) {
        int n = prices.length;
        int profit = 0;

        // iterate through the array
        for (int i = 1; i < n; i++) {
            // if today's price is greater than yesterday's, add the difference
            if (prices[i] > prices[i - 1]) {
                profit += prices[i] - prices[i - 1];
            }
        }
        return profit;
    }
}
